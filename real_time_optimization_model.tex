% =============================================================================
\section*{Terminology}
\addcontentsline{toc}{section}{Terminology}
% =============================================================================

A \textbf{job} refers to any containerized workload unit submitted by a tenant
for execution on the cluster. In Kubernetes this corresponds to a \textbf{pod}
--- the smallest schedulable unit. We use \emph{job} for generality, as the
model is scheduler-agnostic and not specific to Kubernetes.

% =============================================================================
\section{Inputs to Optimizer}
% =============================================================================

The cluster monitor provides real-time inputs to the optimizer each scheduling
round and receives the placement assignment $x_{jn}$ in return.

\medskip
\noindent\textbf{Cluster monitor inputs:}
\begin{itemize}[noitemsep, topsep=2pt]
  \item $\Umem$ --- currently used memory per node
  \item $\vbarsla$ --- rolling SLA violation rate per node (last $K$ rounds)
  \item $\omdelay$ --- scheduling delay weight per job (inherited from tenant;
    computed over last $K$ rounds)
\end{itemize}

\medskip
\noindent\textbf{Prediction layer inputs:}
\begin{itemize}[noitemsep, topsep=2pt]
  \item $\Pmem$ --- maximum predicted memory per pending job over its lifetime
  \item $\Pcpu$ --- P95 predicted CPU peak per pending job over its lifetime
\end{itemize}

\medskip
The optimizer maximizes $Z$ (Placement Score) and outputs $x_{jn}$, which
the cluster monitor uses to submit jobs to nodes.

% =============================================================================
\section{Decision Variable}
% =============================================================================

\[
  x_{jn} \;\in\; \{0,1\}
\]

$x_{jn} = 1$ if job $j$ is assigned to node $n$; $x_{jn} = 0$ otherwise.
This placement matrix is the sole output of the optimizer.

% ── Configuration ────────────────────────────────────────────────────────────
\subsection*{Configuration}
\textit{Set by the cluster operator before the simulation; constant at runtime.}

\medskip
\begin{center}
\begin{tabular}{@{}l p{10.5cm}@{}}
\toprule
Symbol & Meaning \\
\midrule
$\Mtax$        & \textbf{Memory tax} --- fixed overhead reserved for the OS,
  kubelet, and monitoring agents; not available to tenant jobs.
  Based on Google cluster-usage traces v3: the Borglet agent reserves
  $\approx 5\%$ of $M_n$, making the effective usable ceiling $\approx 95\%$
  of physical RAM \\[6pt]
$\Mtheta$      & \textbf{Memory safety threshold buffer} --- fixed memory
  reserved per node as protection against runtime memory spikes.
  Default: $\Mtheta = 0.10 \cdot M_n$ (10\% of physical RAM).
  Together with $\Mtax$, defines the hard capacity ceiling $\Mcap$ \\[6pt]
$K$            & \textbf{Rolling window length} --- number of past scheduling
  rounds used to compute both $\vbarsla$ and $\omdelayt$.
  See Appendices~\ref{app:vbar} and~\ref{app:omega} \\[6pt]
$\sigconsolid$ & \textbf{Fixed node consolidation weight} --- static bias that
  steers placement toward lower-indexed nodes, keeping the cluster
  consolidated. Node 0 gets the highest weight; the last node gets 1.
  Constant across all scheduling rounds.
  $\sigconsolid = |N| - n,\quad n = 0, 1, \ldots, |N|-1$

  \smallskip\textit{Note:} A sequencing constraint (node $n$ may only receive
  jobs if node $n{-}1$ is non-empty) would create a recursive dependency the
  MILP cannot resolve in a single solve call. The fixed weight achieves the
  same consolidation bias without constraint coupling. \\
\bottomrule
\end{tabular}
\end{center}

% ── Cluster Monitor Inputs ───────────────────────────────────────────────────
\subsection*{Cluster Monitor Inputs}
\textit{Provided to the optimizer at the start of each scheduling round.}

\medskip
\begin{center}
\begin{tabular}{@{}l p{10.5cm}@{}}
\toprule
Symbol & Meaning \\
\midrule
$\Umem$ & Currently used memory on node $n$, measured in real time.
  Subscript \textit{mem} distinguishes from future CPU usage metrics
  (e.g.\ $U_n^{\text{cpu}}$) \\[6pt]
$\vbarsla$ & \textbf{Rolling SLA violation rate} on node $n$ --- fraction of
  the last $K$ rounds in which $\Umem > \Mcap$.
  $\bar{v}_n^{\text{SLA}} \in [0,1]$.
  $\vbarsla = 0$: no recent violations. \quad
  $\vbarsla = 1$: every recent round was overloaded.
  See Appendix~\ref{app:vbar} \\[6pt]
$\omdelay$ & \textbf{Scheduling delay weight} for job $j$ --- inherited from
  the tenant that owns job $j$ ($\omdelay = \omega_{t(j)}^{\text{delay}}$);
  computed over the last $K$ scheduling rounds. $\omdelay \geq 1$. \quad
  $\omdelay = 1$: no extra weight (tenant wait at or below cluster average).
  $\omdelay > 1$: additional weight; solver increases preference for job $j$
  (real-valued, no maximum). See Appendix~\ref{app:omega} \\
\bottomrule
\end{tabular}
\end{center}

\toprule
Symbol & Meaning \\

$\Pmem$  & \textbf{Maximum predicted memory} for job $j$ --- worst-case peak

  Used in constraint C2 and the objective \\[6pt]

% ── Derived ──────────────────────────────────────────────────────────────────
\subsection*{Derived}
\textit{Computed by the optimizer from the above inputs each scheduling round.}

\medskip
\begin{center}
\begin{tabular}{@{}l p{10.5cm}@{}}
\toprule
Symbol & Meaning \\
\midrule
$\Mcap$  & \textbf{Node capacity} --- physical RAM minus OS tax and safety
  threshold buffer; fixed ceiling for all tenant job memory on node $n$ \\[6pt]
$\Mavail$ & \textbf{Remaining available memory} --- capacity minus current
  usage; may be negative when over-committed \\[6pt]
$\Meff$  & \textbf{Effective remaining memory} --- capacity offered to new
  jobs after SLA violation scaling. Right-hand side of constraint C2 \\[6pt]
$\omutilize$  & \textbf{Memory utilization weight} --- ratio of current used memory
  to node capacity $\Mcap$, mapped to $[1,2]$. Applied in the objective to
  consolidate workloads onto busier nodes.
  See Appendix~\ref{app:umem} \\
\bottomrule
\end{tabular}
\end{center}

\begin{align}
  \Mcap    &\;=\; M_n - \Mtax - \Mtheta \notag \\[4pt]
  \Mavail  &\;=\; \Mcap - \Umem \notag \\[4pt]
  \Meff    &\;=\; \max\!\left(0,\;\; \Mavail \cdot (1 - \vbarsla)\right) \notag
\end{align}

% =============================================================================
\section{Feedback Loops}
% =============================================================================

SLA and fairness are not objectives inside the LP --- they are observed
outcomes tracked by the cluster monitor and fed back as adaptive inputs each
scheduling round.

\paragraph{SLA Feedback $\rightarrow$ Node Capacity.}
When actual memory on node $n$ exceeds $\Mcap$ at runtime, the cluster
monitor records one overload event. It computes $\vbarsla$ over the last $K$
rounds and passes it to the optimizer. As violations persist, $\vbarsla$ rises
and $\Meff$ shrinks --- the node accepts fewer new jobs, reducing the risk of
OOM kills. As violations subside, $\Meff$ recovers.

\paragraph{Fairness Feedback $\rightarrow$ Job Priority.}
The cluster monitor tracks per-job wait times, computes $\omdelayt$ per tenant
over the last $K$ rounds, and passes it to the optimizer. Jobs whose tenant is
waiting above average receive $\omdelay > 1$, so the solver naturally prefers
placing them. Fairness improves as a side effect of weighted maximization.
Variance of $\bar{W}_t$ across tenants (or Gini coefficient) is reported as a
simulation KPI --- not minimized inside the LP.

% =============================================================================
\section{Optimization Objective}
% =============================================================================

Maximize the \textbf{Placement Score} $Z$:

\[
  \max \quad Z \;=\;
  \sum_{j \in J}\; \sum_{n \in N}
    \omdelay \cdot \Pmem \cdot \omutilize \cdot \sigconsolid \cdot x_{jn}
\]

\noindent\textit{Note on $\omdelay$:} $\omdelay = \omega_{t(j)}^{\text{delay}}$,
the delay weight of the tenant that owns job $j$. Writing $\omdelay$ (per-job
subscript $j$) rather than $\omdelayt$ (per-tenant subscript $t$) is the
convention used throughout this model because the objective is indexed over
jobs. Appendix~\ref{app:omega} defines $\omdelayt$ and the equality follows
directly: $\omdelay \;=\; \omdelayt\big|_{t=t(j)}$.

\medskip
Each placed job contributes its maximum predicted memory $\Pmem$, scaled by
three multiplicative weights:
\begin{itemize}[noitemsep, topsep=4pt]
  \item $\omdelay \geq 1$ --- tenant delay weight: jobs from long-waiting
    tenants get higher weight, so the solver prefers placing them.
  \item $\omutilize \in [1,2]$ --- memory consolidation: a job placed on a
    memory-loaded node (high $\Umem / \Mcap$) contributes more, consolidating
    workloads onto fewer machines.
  \item $\sigconsolid \in \{1,\ldots,|N|\}$ --- fixed node bias: node 0 has the
    highest weight, ensuring consolidation even at the first batch when all
    $\omutilize = 1$ (no prior load history).
\end{itemize}

% =============================================================================
\section{Constraints}
% =============================================================================

\paragraph{C1 --- One node per job.}
Each job is assigned to at most one node per scheduling round:
\[
  \sum_{n \in N} x_{jn} \;\leq\; 1 \qquad \forall\, j \in J
\]
$\leq 1$ (not $= 1$): a job may remain unscheduled if no node has sufficient
capacity. Unscheduled jobs return to the queue, accumulating wait time that
raises their $\omdelay$ in subsequent rounds.

\paragraph{C2 --- Memory capacity per node.}
Total maximum predicted memory of jobs assigned to node $n$ may not exceed its
effective remaining capacity:
\[
  \sum_{j \in J} \Pmem \cdot x_{jn} \;\leq\; \Meff \qquad \forall\, n \in N
\]
This is a hard constraint. $\Meff$ incorporates the OS tax, safety threshold
buffer, current usage, and SLA-driven scaling. The model uses the predicted
maximum memory $\Pmem$ rather than the tenant's declared request
(overcommitment). An SLA violation occurs only if actual runtime memory
exceeds even the predicted maximum --- indicating behaviour outside the
training distribution.

\paragraph{C3 --- Binary domain.}
\[
  x_{jn} \in \{0,1\} \qquad \forall\, j \in J,\; n \in N
\]
This makes the problem a Mixed-Integer Linear Program (MILP). For large
instances, $x_{jn} \in [0,1]$ (LP relaxation with rounding) is an acceptable
approximation.

\paragraph{C4 --- Per-pair CPU fitment.}
\[
  x_{jn} = 0 \;\text{ if }\; \Pcpu > C_n
  \qquad \forall\, j \in J,\; n \in N
\]
A job may not be placed on a node whose total core count is less than the
job's P95 predicted CPU peak. CPU cores are time-shared --- the kernel handles
overcommit via throttling. This check blocks only the genuinely infeasible
case where a job's peak CPU demand exceeds the node's hardware ceiling.

% =============================================================================
\section{Complete Model}
% =============================================================================

\[
  \max \quad Z \;=\;
  \sum_{j \in J}\; \sum_{n \in N}
    \omdelay \cdot \Pmem \cdot \omutilize \cdot \sigconsolid \cdot x_{jn}
\]

\noindent\textbf{Subject to:}
\begin{align*}
  \text{C1:}&\quad \sum_{n \in N} x_{jn} \leq 1
    && \forall\, j \in J \\[4pt]
  \text{C2:}&\quad \sum_{j \in J} \Pmem \cdot x_{jn} \leq \Meff
    && \forall\, n \in N \\[4pt]
  \text{C3:}&\quad x_{jn} \in \{0,1\}
    && \forall\, j \in J,\; n \in N \\[4pt]
  \text{C4:}&\quad x_{jn} = 0 \;\text{ if }\; \Pcpu > C_n
    && \forall\, j \in J,\; n \in N
\end{align*}

\noindent\textbf{Where:}
\begin{align*}
  \Mcap    &\;=\; M_n - \Mtax - \Mtheta \\[4pt]
  \Mavail  &\;=\; \Mcap - \Umem \\[4pt]
  \Meff    &\;=\; \max\!\left(0,\;\; \Mavail \cdot (1 - \vbarsla)\right) \\[4pt]
  \sigconsolid &\;=\; |N| - n, \quad n = 0,\,1,\,\ldots,\,|N|-1
\end{align*}

\section{How $\bar{v}_n^{\text{SLA}}$ is Computed}
\label{app:vbar}

\noindent\textbf{Additional symbols:}

\begin{center}
\begin{tabular}{@{}l p{10.5cm}@{}}
\toprule
Symbol & Meaning \\
\midrule
$K$ & Rolling window length --- number of past scheduling rounds included
  in the violation rate. Configuration scalar \\[6pt]
$k$ & Summation index, $k = 0, 1, \ldots, K{-}1$. $k=0$ is the most recent
  round; $k = K{-}1$ is the oldest \\[6pt]
$\mathbf{1}[\cdot]$ & Indicator function: 1 if the condition holds, 0
  otherwise \\
\bottomrule
\end{tabular}
\end{center}

\medskip
At scheduling round $r$:

\[
  \bar{v}_n^{\text{SLA}} \;=\;
  \frac{1}{K} \sum_{k=0}^{K-1}
    \mathbf{1}\!\left[\, \Umem > \Mcap \;\text{ at round } r - k \,\right]
\]

$\vbarsla$ is the fraction of the last $K$ rounds in which actual memory on
node $n$ exceeded $\Mcap$ (the capacity ceiling after OS tax and safety
threshold buffer). When $\vbarsla$ rises, $\Meff$ shrinks, throttling new job
placements and reducing the risk of OOM kills. It is passed to the optimizer
as an input each round by the cluster monitor.

% =============================================================================
\section{How $\omutilize$ is Computed}
\label{app:umem}
% =============================================================================

\[
  \omutilize \;=\; 1 \;+\; \min\!\left(1,\;\max\!\left(0,\;
    \frac{\Umem}{\max\!\left(1,\, \Mcap\right)}\right)\right)
\]

The denominator is $\Mcap$ (schedulable capacity after OS tax and safety
threshold), so the weight reflects how loaded the node is relative to the
memory actually available for tenant jobs. When a node is fully packed up to
$\Mcap$, the weight reaches 2.

\begin{itemize}[noitemsep, topsep=4pt]
  \item $\omutilize = 1$: node is idle (no consolidation boost)
  \item $\omutilize = 2$: node is fully packed to $\Mcap$ (maximum consolidation boost)
  \item $1 < \omutilize < 2$: partial load (real-valued and continuous)
\end{itemize}

% =============================================================================
\section{How $\omdelayt$ is Computed}
\label{app:omega}
% =============================================================================

$\omdelayt$ is the scheduling delay weight for tenant $t$, computed by the
cluster monitor each round and inherited by every job $j$ where $t(j) = t$
(i.e.\ $\omdelay = \omdelayt\big|_{t=t(j)}$). Both $\bar{W}_t$ and $\bar{W}$
are computed over the last $K$ scheduling rounds (same rolling window as
$\vbarsla$), so the weight reflects \emph{recent} fairness rather than
all-time history.

\medskip
\noindent\textbf{Additional symbols:}

\begin{center}
\begin{tabular}{@{}l p{10.5cm}@{}}
\toprule
Symbol & Meaning \\
\midrule
$\bar{W}_t$ & Average scheduling delay for tenant $t$ --- mean wait time from
  job submission to job assignment, over the last $K$ scheduling rounds \\[6pt]
$\bar{W}$ & Cluster-wide average scheduling delay over the last $K$ rounds:
  $\displaystyle\bar{W} = \frac{1}{|T|}\sum_{t \in T} \bar{W}_t$ \\
\bottomrule
\end{tabular}
\end{center}

\[
  \omdelayt \;=\; 1 \;+\; \max\!\left(0,\;
    \frac{\bar{W}_t - \bar{W}}{\max\!\left(1,\, \bar{W}\right)}
    \right)
\]

\[
  \omdelay \;=\; \omdelayt\big|_{t\,=\,t(j)}
\]

The weight is built from the excess wait above the cluster average. Adding 1
ensures the minimum is 1 (no unfairness = no extra weight). There is no stated
maximum: a tenant waiting far above average receives $\omdelayt \gg 1$, giving
their jobs proportionally higher priority.

\begin{itemize}[noitemsep, topsep=4pt]
  \item $\omdelay = 1$ when $\bar{W}_{t(j)} \leq \bar{W}$ --- no boost
  \item $\omdelay > 1$ when $\bar{W}_{t(j)} > \bar{W}$ --- boost grows
    proportionally with excess wait (real-valued, no maximum)
\end{itemize}
