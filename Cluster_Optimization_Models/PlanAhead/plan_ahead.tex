% ─────────────────────────────────────────────────────────────
\section{Overview}
% ─────────────────────────────────────────────────────────────

The plan-ahead model is a MISOCP solved periodically (every
\texttt{plan\_ahead\_interval} simulation steps).  Its purpose is to
determine which nodes each tenant should be \emph{prioritised} on for
each planning period, given estimated resource demand, while providing
a probabilistic guarantee that node capacity is not exceeded.

The output is a \textbf{priority hint} --- a
\texttt{TenantAccessSchedule} mapping $(\text{tenant}, \text{period})
\to [\text{node\_ids}]$.  The real-time scheduler uses this hint to
boost objective coefficients; no node is blocked.

The second-order cone component (C1b) implements a
\textbf{variance-extended Cantelli capacity constraint}: it ensures
that node capacity is respected with at least probability $1-\varepsilon$
even under demand uncertainty, using only the per-tenant usage profile
variance --- not per-workload covariance as in the original formulation.

\section{Parameters}
% ─────────────────────────────────────────────────────────────

\begin{longtable}{@{}lp{10cm}@{}}
  \toprule
  \textbf{Symbol} & \textbf{Description} \\
  \midrule
  $u_{i,h}$ & Estimated total resource demand of tenant $i$ in period $h$ (capacity units). Placeholder for prediction layer output. \\[2pt]
  $C_n$     & Resource capacity of node $n$ (uniform in synthetic data). \\[2pt]
  $\pi_n$   & Infrastructure cost of activating node $n$ per period. \\[2pt]
  $\bar\pi_i$ & Contract revenue for admitting tenant $i$. \\[2pt]
  $v_i$     & Operational cost for tenant $i$. \\[2pt]
  $\lambda_0, \lambda_1, \lambda_2$ & Objective weights (infrastructure cost, admission revenue, fairness). \\[4pt]
  \midrule
  \multicolumn{2}{@{}l}{\textit{Cantelli uncertainty parameters}} \\[2pt]
  \texttt{sigma\_frac} & Demand uncertainty fraction: $\mathrm{std}(D_{i,h}) = \texttt{sigma\_frac} \times u_{i,h}$. \\[2pt]
  $\sigma^2_{i,h}$ & Per-tenant per-period demand variance: $(\texttt{sigma\_frac} \times u_{i,h})^2$. \\[2pt]
  $\varepsilon$ & Cantelli tail probability (\texttt{cantelli\_epsilon}); default 0.10. \\[2pt]
  $\kappa$ & Cantelli safety factor: $\kappa = \sqrt{(1-\varepsilon)/\varepsilon}$. With $\varepsilon=0.10$, $\kappa = 3.0$. \\
  \bottomrule
\end{longtable}

\subsection*{Tenant Usage Profiles}

$u_{i,h}$ is drawn from \texttt{Uniform(tenant\_usage\_min, tenant\_usage\_max)}
as a placeholder.  In production, derive from historical
CollectionEvents SUBMIT counts per tenant per access period.

\subsection*{Uncertainty Model}

The demand of tenant $i$ on node $n$ in period $h$ is modelled as a random
variable $D_{i,n,h}$ with mean $f_{i,n,h}$ and variance
$\sigma^2_{i,h} \cdot y_{i,n,h}$.  The variance is active only when the
tenant is assigned to that node ($y_{i,n,h}=1$).  By the \emph{one-sided
Cantelli inequality}:

\[
  P\!\left[\sum_{i \in \cT} D_{i,n,h} \leq C_n \cdot z_{n,h}\right] \;\geq\; 1 - \varepsilon
\]

is guaranteed by C1a and C1b together (see Section~\ref{sec:constraints}).

\section{Decision Variables}
% ─────────────────────────────────────────────────────────────

\begin{longtable}{@{}llp{9cm}@{}}
  \toprule
  \textbf{Variable} & \textbf{Domain} & \textbf{Description} \\
  \midrule
  $f_{i,n,h}$ & $\geq 0$ & Allocation of node $n$'s capacity to tenant $i$ in period $h$. \\[2pt]
  $y_{i,n,h}$ & $\{0,1\}$ & 1 iff tenant $i$ has a priority assignment on node $n$ in period $h$. \\[2pt]
  $z_{n,h}$   & $\{0,1\}$ & 1 iff node $n$ is active in period $h$. \\[2pt]
  $t_{n,h}$   & $\geq 0$  & Cantelli slack --- upper bound on weighted demand standard deviation on node $n$ in period $h$. \\[2pt]
  $a_i$       & $\{0,1\}$ & 1 iff tenant $i$ is admitted for this planning horizon. \\[2pt]
  $\sigma$    & $[0,1]$   & Minimum demand-satisfaction ratio across admitted tenants (fairness proxy). \\
  \bottomrule
\end{longtable}

% ─────────────────────────────────────────────────────────────
\section{Constraints}
\label{sec:constraints}
% ─────────────────────────────────────────────────────────────

\subsection*{C1a --- Capacity with Cantelli Margin (linear)}

\begin{equation}
  \sum_{i \in \cT} f_{i,n,h} + \kappa \cdot t_{n,h} \;\leq\; C_n \cdot z_{n,h}
  \qquad \forall\, n \in \cN,\; h \in \cH
  \tag{C1a}
\end{equation}

Mean allocations plus a $\kappa$-weighted uncertainty margin must not exceed
node capacity when active.  When $\kappa = 0$ (no uncertainty), this reduces
to the deterministic capacity constraint.

\subsection*{C1b --- Second-Order Cone Constraint (SOCP)}

\begin{equation}
  \sum_{i \in \cT} \sigma^2_{i,h} \cdot y_{i,n,h} \;\leq\; t_{n,h}^2
  \qquad \forall\, n \in \cN,\; h \in \cH
  \tag{C1b}
\end{equation}

Forces $t_{n,h}$ to be at least the weighted demand standard deviation on
node $n$ in period $h$.  This is a second-order cone constraint and is what
makes the model an MISOCP.  In Gurobi it is expressed as a
\texttt{QConstr}: \mbox{$t_{n,h}^2 \geq \sum_i \sigma^2_{i,h} \cdot y_{i,n,h}$}.

\medskip
\noindent\textit{Combined probabilistic guarantee:} C1a and C1b together
ensure $P[\sum_i D_{i,n,h} \leq C_n z_{n,h}] \geq 1-\varepsilon$
with $\varepsilon = 0.10$, $\kappa = 3.0$ by default.

\subsection*{C2 --- Priority Link}

\begin{equation}
  f_{i,n,h} \;\leq\; C_n \cdot y_{i,n,h}
  \qquad \forall\, i \in \cT,\; n \in \cN,\; h \in \cH
  \tag{C2}
\end{equation}

If $y_{i,n,h}=0$, no capacity can be allocated to tenant $i$ on node $n$
in period $h$.  This links the continuous allocation to the binary priority
indicator.

\subsection*{C3 --- Demand Satisfaction}

\begin{equation}
  \sum_{n \in \cN} f_{i,n,h} \;\geq\; u_{i,h} \cdot a_i
  \qquad \forall\, i \in \cT,\; h \in \cH
  \tag{C3}
\end{equation}

Admitted tenants must receive at least their estimated demand in each period.
Rejected tenants ($a_i=0$) have this constraint deactivated.

\subsection*{C4 --- Fairness}

\begin{equation}
  \sigma \;\leq\;
  \frac{\displaystyle\sum_{n \in \cN,\, h \in \cH} f_{i,n,h}}
       {\displaystyle\sum_{h \in \cH} u_{i,h}}
  + (1 - a_i)
  \qquad \forall\, i \in \cT
  \tag{C4}
\end{equation}

$\sigma$ is the minimum demand-satisfaction ratio across all admitted tenants
(DRF-style fairness).  The $(1-a_i)$ term deactivates this constraint for
rejected tenants.  Since $u_{i,h}$ is a parameter, this constraint is
\textbf{linear}.

\subsection*{C5 --- Node Activation}

\begin{equation}
  z_{n,h} \;\geq\; y_{i,n,h}
  \qquad \forall\, i \in \cT,\; n \in \cN,\; h \in \cH
  \tag{C5}
\end{equation}

A node must be marked active if any tenant has a priority assignment on it.

% ─────────────────────────────────────────────────────────────
\section{Objective Function}
% ─────────────────────────────────────────────────────────────

\begin{equation}
  \min\quad
  \underbrace{\lambda_0 \sum_{n \in \cN,\, h \in \cH} \pi_n \cdot z_{n,h}}_{\text{infrastructure cost}}
  - \underbrace{\lambda_1 \sum_{i \in \cT} (\bar\pi_i - v_i) \cdot a_i}_{\text{admission revenue}}
  - \underbrace{\lambda_2 \cdot \sigma}_{\text{fairness}}
  \tag{OBJ}
\end{equation}

The three components balance:
\begin{enumerate}[leftmargin=2cm]
  \item \textbf{Infrastructure cost} --- prefer fewer active nodes.
  \item \textbf{Admission revenue} --- prefer admitting more tenants.
  \item \textbf{Fairness} --- maximise the minimum demand-satisfaction ratio.
\end{enumerate}

% ─────────────────────────────────────────────────────────────
\section{Output Interface}
% ─────────────────────────────────────────────────────────────

\texttt{extract\_tenant\_access\_schedule(vars\_, P)} returns:
\[
  \texttt{TenantAccessSchedule} = \bigl\{(i,\,h) \;\mapsto\; \{n : y_{i,n,h}=1\}\bigr\}
\]

These are \textbf{priority hints}.  The real-time scheduler applies a
\texttt{PRIORITY\_BOOST} $= 2.0$ multiplier to the objective coefficient
for $(j,\,n)$ pairs where $n \in \texttt{TenantAccessSchedule}[t(j),\,h]$,
and uses \texttt{sort\_by\_plan\_priority} to front-load prioritised jobs
in the solver window.  \emph{All nodes remain accessible regardless.}

% ─────────────────────────────────────────────────────────────
\section{Real-Time Integration}
% ─────────────────────────────────────────────────────────────

The real-time MILP maximises:
\[
  \max \sum_{j \in J} \sum_{n \in \cN}
  \omega_{\delta,t(j)} \cdot \hat{m}_j \cdot u_n^{\mathrm{mem}} \cdot \sigma_n^{\mathrm{consolid}}
  \cdot b_{t(j),n} \cdot x_{jn}
\]
where
\[
  b_{t(j),n} =
  \begin{cases}
    \text{PRIORITY\_BOOST} & n \in \text{TenantAccessSchedule}[t(j), h] \\
    1.0                    & \text{otherwise}
  \end{cases}
\]

This replaces the former hard-block constraint
$x_{jn} = 0 \text{ if } n \notin A_{t(j)}$ with a soft objective preference.
